%------------------------------------------------------------------------------
%% new commands
%------------------------------------------------------------------------------
\newcommand{\topLegend}[2]{
  node [text centered,%
  shade,%
  top color=\fColor,%
  text justified,
  draw,%
  semithick,
  rounded corners=2pt,
  above,
  #1] {#2}}

\newcommand{\bottomLegend}[2]{
  node [text centered,%
  shade,%
  top color=\fColor,%
  text justified,
  draw,%
  semithick,
  rounded corners=2pt,
  below,
  #1] {#2}}

\newcommand{\topEvent}[3]{\draw[<-,very thick] (#1) coordinate (#2) (#2 |- xH axis) -- ++ (#3)}
\newcommand{\bottomEvent}[3]{\draw[<-,very thick] (#1) coordinate (#2) (#2 |- xL axis) -- ++ (#3)}


\newcommand{\bottomGraphics}[2]{node[draw,anchor=north,inner sep=1px,thin] {\includegraphics[#1]{#2}}}
\newcommand{\topGraphics}[2]{node[draw,anchor=south,inner sep=1px,thin] {\includegraphics[#1]{#2}}}
